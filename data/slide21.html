    Text Formating:
    • 'HTML Text formatting' -
      • '< b >', '< strong >' - bold
      • ' < em >', < i > - italic
      • ' < u > ' - underline
      • ' < s > ' - strike-through
      • ' < small > ' - smaller font
      • ' < big > ' - larger font
      • ' < acronym title="Подсказка" > Tip < / acronym > ' - tooltip
      • ' < blockquote > Цитата < / blockquote > ' - quotes formatting
      • ' < pre > ' - the text will be represented as it stands in the raw HTML, no matter the white-spaces [usually used for code highlights]
      • ' < marquee behavior="alternate" direction="right" class="myText" >Hello world! < / marquee > ' - running line of text rightwards
      • ' < a href=”*” target=”_blank” title=”Подсказка” > < img src=”*”>< / a> ' - anchor-wrapper for an image with a tooltip
      • ' < a href = “#anchor1”> Якорь< / a> - link to the 'anchor1' '
    • 'CSS text formatting' -
      • ' font-size: '
      • ' font-family: '
      • ' color: '
      • ' background-color: '
      • ' text-shadow: 2px 2px 1px green; ' setting the shadow (offset by X, Y; the degree of the shadow; color;)
      • ' text - decoration: none; ' is widely used for links to strip default styling
      • ' font-style: italic; ' - style options: (normal, italic, oblique)
      • ' font-weight: bold; ' - weight options: (normal, bold, bolder, lighter)
      • ' text - transform: ' - text capitalization control (none|capitalize|uppercase|lowercase|initial|inherit),
    • 'Relative units of measurements' -
      • ' em ' depends on the parent, the current browser style of the font
      • ' ex ' the height of the 'x' symbol
      • ' % ' depends on the parent, the current browser style of the font
      • ' rem ' font height, which does not depend on the parent
    • 'Absolute units of measurements' -
      • ' in ' inch
      • ' cm '
      • ' mm '
      • ' pt ' 1/72 of an inch
      • ' pc ' 12 points
      • ' px '
    • 'Adding custom fonts:' - google Fonts
      @font-face {
      font-family: font name;
      src: url(“path to the sourse font file”);
      }
